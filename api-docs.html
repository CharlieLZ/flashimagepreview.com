üöÄ API Overview

The Flash Image Preview API provides powerful image processing capabilities through a simple RESTful interface. Whether you're building a photo editor, image gallery, or any application that needs professional image processing, our API offers the tools you need.

‚ö° Fast Processing

Process images in milliseconds with our optimized infrastructure

üîí Secure & Private

Enterprise-grade security with optional client-side processing

üì± Universal Support

Works with all major image formats and platforms

üéØ Developer Friendly

Simple REST API with comprehensive SDKs and documentation

Base URL

https://api.flashimagepreview.com/v1

Supported Formats

Input Formats

  • JPEG, JPG
  • PNG (with transparency)
  • GIF (animated support)
  • WebP
  • AVIF
  • TIFF
  • BMP
  • SVG

Output Formats

  • JPEG (optimized)
  • PNG (lossless)
  • WebP (modern)
  • AVIF (next-gen)
  • PDF (documents)

üîê Authentication

All API requests require authentication using API keys. You can obtain your API key from your dashboard after creating an account.

API Key Authentication

Include your API key in the Authorization header of every request:

curl -H "Authorization: Bearer YOUR_API_KEY" \
     -H "Content-Type: application/json" \
     https://api.flashimagepreview.com/v1/process

JavaScript SDK Authentication

const flashAPI = new FlashImagePreview({
  apiKey: 'YOUR_API_KEY',
  environment: 'production' // or 'sandbox'
});

üõ°Ô∏è Security Best Practices

  • Never expose your API key in client-side code
  • Use environment variables for API key storage
  • Rotate your API keys regularly
  • Use different keys for development and production

üöÄ Quick Start Guide

Get started with the Flash Image Preview API in just a few minutes. This guide will walk you through uploading and processing your first image.

Step 1: Upload an Image

curl -X POST https://api.flashimagepreview.com/v1/images \
     -H "Authorization: Bearer YOUR_API_KEY" \
     -F "image=@/path/to/your/image.jpg" \
     -F "name=my-test-image"

Response:

{
  "id": "img_abc123",
  "name": "my-test-image",
  "format": "jpeg",
  "width": 1920,
  "height": 1080,
  "size": 245760,
  "url": "https://cdn.flashimagepreview.com/img_abc123.jpg",
  "created_at": "2025-01-15T10:30:00Z"
}

Step 2: Process the Image

curl -X POST https://api.flashimagepreview.com/v1/images/img_abc123/process \
     -H "Authorization: Bearer YOUR_API_KEY" \
     -H "Content-Type: application/json" \
     -d '{
       "operations": [
         {"type": "resize", "width": 800, "height": 600},
         {"type": "filter", "name": "brightness", "value": 0.1},
         {"type": "format", "output": "webp", "quality": 90}
       ]
     }'

Response:

{
  "id": "proc_xyz789",
  "status": "completed",
  "output": {
    "id": "img_def456",
    "format": "webp",
    "width": 800,
    "height": 600,
    "size": 67890,
    "url": "https://cdn.flashimagepreview.com/img_def456.webp"
  },
  "processing_time": 234
}

Step 3: Download the Result

Use the returned URL to download or display the processed image:

https://cdn.flashimagepreview.com/img_def456.webp

üì¶ JavaScript SDK

Our JavaScript SDK provides a convenient way to integrate Flash Image Preview into your web applications with full TypeScript support.

Installation

# npm
npm install @flashimagepreview/sdk

# yarn
yarn add @flashimagepreview/sdk

# CDN
<script src="https://cdn.flashimagepreview.com/sdk/v1/flash-image-preview.min.js"></script>

Basic Usage

import FlashImagePreview from '@flashimagepreview/sdk';

const flash = new FlashImagePreview({
  apiKey: 'YOUR_API_KEY'
});

// Upload and process an image
async function processImage(file) {
  try {
    // Upload image
    const uploadResult = await flash.images.upload(file, {
      name: 'user-photo'
    });
    
    // Process image
    const processResult = await flash.images.process(uploadResult.id, {
      operations: [
        { type: 'resize', width: 400, height: 400, mode: 'cover' },
        { type: 'filter', name: 'saturation', value: 0.2 },
        { type: 'format', output: 'webp', quality: 85 }
      ]
    });
    
    return processResult.output.url;
  } catch (error) {
    console.error('Processing failed:', error);
  }
}

React Hook Example

import { useState } from 'react';
import { useFlashImagePreview } from '@flashimagepreview/react';

function ImageProcessor() {
  const [processedUrl, setProcessedUrl] = useState(null);
  const { processImage, isProcessing } = useFlashImagePreview();

  const handleFileSelect = async (event) => {
    const file = event.target.files[0];
    if (!file) return;

    const result = await processImage(file, {
      resize: { width: 800, height: 600 },
      filters: [
        { name: 'brightness', value: 0.1 },
        { name: 'contrast', value: 0.05 }
      ],
      format: { output: 'webp', quality: 90 }
    });

    setProcessedUrl(result.url);
  };

  return (
    <div>
      <input type="file" onChange={handleFileSelect} accept="image/*" />
      {isProcessing && <p>Processing...</p>}
      {processedUrl && <img src={processedUrl} alt="Processed" />}
    </div>
  );
}

üí° Code Examples

Complete Image Processing Example

async function processUserPhoto(file) {
  const API_KEY = 'your_api_key_here';
  const BASE_URL = 'https://api.flashimagepreview.com/v1';

  try {
    // Upload image
    const formData = new FormData();
    formData.append('image', file);
    formData.append('name', 'user-photo');

    const uploadResponse = await fetch(`${BASE_URL}/images`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${API_KEY}`
      },
      body: formData
    });

    const uploadResult = await uploadResponse.json();
    
    // Process image
    const processResponse = await fetch(`${BASE_URL}/images/${uploadResult.id}/process`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${API_KEY}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        operations: [
          { type: 'resize', width: 800, height: 600, mode: 'cover' },
          { type: 'filter', name: 'brightness', value: 0.1 },
          { type: 'format', output: 'webp', quality: 85 }
        ]
      })
    });

    const processResult = await processResponse.json();
    return processResult.output.url;
    
  } catch (error) {
    console.error('Image processing failed:', error);
    throw error;
  }
}

Python Image Processing

import requests
import json

def process_image(image_path, api_key):
    base_url = "https://api.flashimagepreview.com/v1"
    headers = {"Authorization": f"Bearer {api_key}"}
    
    # Upload image
    with open(image_path, 'rb') as f:
        files = {'image': f}
        data = {'name': 'python-upload'}
        
        upload_response = requests.post(
            f"{base_url}/images",
            headers=headers,
            files=files,
            data=data
        )
        
    upload_result = upload_response.json()
    image_id = upload_result['id']
    
    # Process image
    process_data = {
        "operations": [
            {"type": "resize", "width": 1200, "height": 800, "mode": "cover"},
            {"type": "filter", "name": "saturation", "value": 0.2},
            {"type": "format", "output": "webp", "quality": 90}
        ]
    }
    
    process_response = requests.post(
        f"{base_url}/images/{image_id}/process",
        headers={**headers, "Content-Type": "application/json"},
        data=json.dumps(process_data)
    )
    
    return process_response.json()['output']['url']

# Usage
api_key = "your_api_key_here"
result_url = process_image("photo.jpg", api_key)
print(f"Processed image: {result_url}")

PHP Implementation

<?php

class FlashImagePreview {
    private $apiKey;
    private $baseUrl = 'https://api.flashimagepreview.com/v1';
    
    public function __construct($apiKey) {
        $this->apiKey = $apiKey;
    }
    
    public function uploadImage($imagePath, $name = null) {
        $ch = curl_init();
        
        curl_setopt_array($ch, [
            CURLOPT_URL => $this->baseUrl . '/images',
            CURLOPT_POST => true,
            CURLOPT_RETURNTRANSFER => true,
            CURLOPT_HTTPHEADER => [
                'Authorization: Bearer ' . $this->apiKey
            ],
            CURLOPT_POSTFIELDS => [
                'image' => new CURLFile($imagePath),
                'name' => $name ?: basename($imagePath)
            ]
        ]);
        
        $response = curl_exec($ch);
        curl_close($ch);
        
        return json_decode($response, true);
    }
    
    public function processImage($imageId, $operations) {
        $ch = curl_init();
        
        curl_setopt_array($ch, [
            CURLOPT_URL => $this->baseUrl . '/images/' . $imageId . '/process',
            CURLOPT_POST => true,
            CURLOPT_RETURNTRANSFER => true,
            CURLOPT_HTTPHEADER => [
                'Authorization: Bearer ' . $this->apiKey,
                'Content-Type: application/json'
            ],
            CURLOPT_POSTFIELDS => json_encode(['operations' => $operations])
        ]);
        
        $response = curl_exec($ch);
        curl_close($ch);
        
        return json_decode($response, true);
    }
}

// Usage
$flash = new FlashImagePreview('your_api_key_here');

$upload = $flash->uploadImage('photo.jpg', 'test-image');
$process = $flash->processImage($upload['id'], [
    ['type' => 'resize', 'width' => 800, 'height' => 600],
    ['type' => 'filter', 'name' => 'brightness', 'value' => 0.1],
    ['type' => 'format', 'output' => 'webp', 'quality' => 85]
]);

echo "Processed image URL: " . $process['output']['url'];

?>

Node.js with Express

const express = require('express');
const multer = require('multer');
const FormData = require('form-data');
const fetch = require('node-fetch');

const app = express();
const upload = multer({ dest: 'uploads/' });

const API_KEY = 'your_api_key_here';
const BASE_URL = 'https://api.flashimagepreview.com/v1';

app.post('/process-image', upload.single('image'), async (req, res) => {
  try {
    // Upload to Flash Image Preview
    const form = new FormData();
    form.append('image', fs.createReadStream(req.file.path));
    form.append('name', req.file.originalname);

    const uploadResponse = await fetch(`${BASE_URL}/images`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${API_KEY}`,
        ...form.getHeaders()
      },
      body: form
    });

    const uploadResult = await uploadResponse.json();

    // Process the image
    const processResponse = await fetch(`${BASE_URL}/images/${uploadResult.id}/process`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${API_KEY}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        operations: [
          { type: 'resize', width: 1200, height: 800, mode: 'cover' },
          { type: 'filter', name: 'contrast', value: 0.1 },
          { type: 'format', output: 'webp', quality: 90 }
        ]
      })
    });

    const processResult = await processResponse.json();
    
    // Clean up uploaded file
    fs.unlinkSync(req.file.path);
    
    res.json({
      success: true,
      url: processResult.output.url
    });

  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

app.listen(3000, () => {
  console.log('Server running on port 3000');
});
